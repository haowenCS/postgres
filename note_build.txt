mkdir build
cd build
../configure --prefix=$PWD
sudo make -j 12
sudo make install

# create the data directory
mkdir data
./bin/initdb ./data/

# start the server
./bin/pg_ctl -D ./data/ start

# stop the server
./bin/pg_ctl stop -D ./data -m fast

# configure file location: ./data/postgresql.conf

# create a default user
./bin/createuser -s -r postgres

# connect to server
./bin/psql postgres

\du

\c testdb # connect
